# Azure B2C Authentication Configuration
# Copy these variables to your .env.local file and replace the placeholder values

# Client ID from your Azure B2C app registration
NEXT_PUBLIC_AD_PUBLIC_CLIENT_ID="your-client-id-here"

# Authority for Vipps SSO user flow (used when clicking "Continue with Vipps")
NEXT_PUBLIC_AD_PUBLIC_AUTHORITY="https://your-tenant.b2clogin.com/your-tenant.onmicrosoft.com/B2C_1_signup_signin_vipps"

# Redirect URL after successful authentication (auth callback page)
# Note: Using non-locale route for better reliability

# Redirect URL after logout
NEXT_PUBLIC_AD_PUBLIC_POST_LOGOUT_REDIRECT_URL="http://localhost:3000"

# Known authority for token validation (your B2C tenant domain)
NEXT_PUBLIC_AD_PUBLIC_KNOWN_AUTHORITY="https://your-tenant.b2clogin.com"
